Diagnosing Unadjusted Fits and Constructing the Trimmed Dataset

Author

Siard van den Bosch

Published

January 22, 2025

Modified

Purpose Diagnose unadjusted fits and build a sensitivity-filtered dataset
Inputs unadjusted_data.rds, unadjusted_fits.rds
Outputs adjusted_data.rds, unadjusted_influence_results.rds
Method MM-estimation residual diagnostics; OLS companion tests; multi-criterion influence flagging
Requires Step 7 (Unadjusted Models)
NoteRationale
  • MM-estimation as primary estimator: bounded influence and high efficiency without strict distributional assumptions (Yohai 1987)
  • Descriptive residual summaries over normality tests: large-sample normality tests reject trivial departures (Maronna et al. 2006)
  • OLS companions for classical tests: enables transparent reporting of Breusch-Pagan, RESET, and Durbin-Watson on identical row sets (Koenker 1981)
  • Combined-criteria influence flagging over automatic deletion: MM-estimators already downweight outliers via bounded influence (Hampel et al. 1986)
  • Cross-model aggregation of flags: targets cases repeatedly problematic across nested specifications (Cook and Weisberg 1982)
  • Sensitivity dataset as robustness check: full and filtered fits are compared, not corrected away (Rousseeuw and Leroy 1987)

# compute sample skewness and kurtosis without extra packages
skew_kurt <- function(x) {
  x <- x[is.finite(x)]
  n <- length(x)
  if (n < 4) {
    return(c(skewness = NA_real_, kurtosis = NA_real_))
  }

  m <- mean(x)
  s <- stats::sd(x)
  if (!is.finite(s) || s <= .Machine$double.eps) {
    return(c(skewness = NA_real_, kurtosis = NA_real_))
  }

  z <- (x - m) / s
  c(
    skewness = mean(z^3),
    kurtosis = mean(z^4)
  )
}

# summarize standardized robust residuals for a single model
residual_shape <- function(fit) {
  r <- stats::residuals(fit) / fit$scale
  r <- r[is.finite(r)]

  if (length(r) == 0) {
    return(data.frame(
      n = 0L,
      mean = NA_real_,
      sd = NA_real_,
      median = NA_real_,
      MAD = NA_real_,
      skewness = NA_real_,
      kurtosis = NA_real_,
      p_gt2 = NA_real_,
      p_gt3 = NA_real_
    ))
  }

  sk <- skew_kurt(r)

  data.frame(
    n = length(r),
    mean = mean(r),
    sd = stats::sd(r),
    median = stats::median(r),
    MAD = stats::mad(r),
    skewness = unname(sk["skewness"]),
    kurtosis = unname(sk["kurtosis"]),
    p_gt2 = mean(abs(r) > 2),
    p_gt3 = mean(abs(r) > 3)
  )
}

# compute VIF from model matrix
calc_vif <- function(mm) {
  int <- match("(Intercept)", colnames(mm), nomatch = 0L)
  if (int > 0L) {
    mm <- mm[, -int, drop = FALSE]
  }

  if (ncol(mm) < 2L) {
    return(setNames(numeric(0), character(0)))
  }

  ok <- rowSums(!is.finite(mm)) == 0L
  mm <- mm[ok, , drop = FALSE]

  if (nrow(mm) < 3L) {
    out <- rep(NA_real_, ncol(mm))
    names(out) <- colnames(mm)
    return(out)
  }

  vifs <- vapply(seq_len(ncol(mm)), function(j) {
    y <- mm[, j]
    x <- mm[, -j, drop = FALSE]

    sst <- sum((y - mean(y))^2)
    if (!is.finite(sst) || sst <= .Machine$double.eps) {
      return(NA_real_)
    }

    aux <- stats::lm.fit(x = cbind(1, x), y = y)
    rss <- sum(aux$residuals^2)
    r2 <- 1 - rss / sst
    r2 <- max(min(r2, 1), 0)

    if (r2 >= 1 - sqrt(.Machine$double.eps)) {
      return(Inf)
    }

    1 / (1 - r2)
  }, numeric(1))

  names(vifs) <- colnames(mm)
  vifs
}

# extract maximum VIF from a fitted model
get_max_vif <- function(fit) {
  mm <- tryCatch(stats::model.matrix(fit), error = function(e) NULL)
  if (is.null(mm)) {
    return(NA_real_)
  }

  vifs <- calc_vif(mm)
  if (length(vifs) == 0 || all(is.na(vifs))) {
    return(NA_real_)
  }

  max(vifs, na.rm = TRUE)
}

# extract participant IDs from fitted model
get_pid <- function(fit) {
  mf <- tryCatch(stats::model.frame(fit), error = function(e) NULL)
  pid <- if (!is.null(mf)) rownames(mf) else NULL

  if (is.null(pid) || anyNA(pid) || any(pid == "")) {
    pid <- as.character(seq_along(stats::residuals(fit)))
  }

  pid
}

# create long-format influence diagnostics for a single model
influence_long_df <- function(fit, model_name) {
  pid <- get_pid(fit)
  n <- length(pid)

  r <- stats::residuals(fit) / fit$scale
  h <- tryCatch(stats::hatvalues(fit), error = function(e) rep(NA_real_, n))
  w <- fit$rweights
  if (is.null(w)) {
    w <- rep(NA_real_, n)
  }

  p <- length(stats::coef(fit))
  lev_thr <- 2 * p / n

  score <- abs(r) * (h / pmax(1 - h, .Machine$double.eps))

  tibble::tibble(
    pid = as.character(pid),
    Model = model_name,
    std_resid = as.numeric(r),
    leverage = as.numeric(h),
    rweight = as.numeric(w),
    lev_thr = lev_thr,
    score = as.numeric(score)
  )
}

# map flagged pids back to data rows
match_ids_to_rows <- function(data, pids) {
  pids <- unique(as.character(pids))
  if (length(pids) == 0) {
    return(integer(0))
  }

  candidate_cols <- c("pid", "id", "obs_id", "row_id", "case_id", "participant_id", "idnummer")
  data_cols_lower <- tolower(names(data))

  for (nm in candidate_cols) {
    j <- match(nm, data_cols_lower, nomatch = 0L)
    if (j > 0L) {
      col <- names(data)[j]
      hits <- which(as.character(data[[col]]) %in% pids)
      if (length(hits) > 0) {
        return(hits)
      }
    }
  }

  if (!is.null(rownames(data))) {
    hits <- which(rownames(data) %in% pids)
    if (length(hits) > 0) {
      return(hits)
    }

  breaks <- unique(stats::quantile(df$influence, c(0, 0.33, 0.67, 1), na.rm = TRUE))
  breaks <- breaks[is.finite(breaks)]
  n_intervals <- length(breaks) - 1

  if (n_intervals <= 0) {
    df$influence_cat <- factor("Low", levels = "Low")
  } else {
    labels <- c("Low", "Medium", "High")[seq_len(n_intervals)]
    df$influence_cat <- cut(
      df$influence,
      breaks = breaks,
      labels = labels,
      include.lowest = TRUE
    )
  }

  p <- ggplot2::ggplot(df, ggplot2::aes(leverage, std_resid, color = influence_cat)) +
    ggplot2::geom_point(alpha = 0.6) +
    ggplot2::geom_smooth(
      method = "loess",
      se = TRUE,
      alpha = 0.3,
      formula = y ~ x,
      method.args = list(
        span = 1,
        degree = 1,
        family = "symmetric",
        control = stats::loess.control(surface = "direct")
      )
    ) +
    ggplot2::labs(
      title = name,
      x = "Leverage",
      y = "Standardized Residuals",
      color = "Influence"
    ) +
    ggplot2::scale_color_viridis_d(option = "D", direction = -1) +
    ggplot2::guides(color = ggplot2::guide_legend(title = "Influence"))

  if (requireNamespace("ggrepel", quietly = TRUE) && any(df$flagged, na.rm = TRUE)) {
    p <- p + ggrepel::geom_text_repel(
      ggplot2::aes(label = label),
      size = 2,
      max.overlaps = 20,
      na.rm = TRUE
    )
  }

  p
}
