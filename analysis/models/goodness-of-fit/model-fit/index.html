Purpose Compare goodness-of-fit metrics across unadjusted and adjusted fits.
Inputs unadjusted_fits.rds, adjusted_data.rds, adjusted_fits.rds
Outputs fit_metrics.rds, nested_rlm_unadjusted.rds, nested_rlm_adjusted.rds, nested_ols_unadjusted.rds, nested_ols_adjusted.rds
Method Goodness-of-fit summaries; nested Wald and deviance tests
Requires Step 7 (Unadjusted Models), Step 8 (Influence Diagnostics), Step 9 (Adjusted Models)
NoteRationale
  • Wald and deviance tests: standard frequentist tests for incremental fit in nested specifications (Burnham and Anderson 2004)
  • MM-estimation: high breakdown point combined with asymptotic efficiency (Yohai 1987)
  • Model 0 retained in RLM comparisons: provides a robust-scale baseline for nested tests.
  • Model 0 excluded from OLS comparisons: variance comparisons against an intercept-only model are uninformative.

# extract goodness-of-fit metrics from model object
get_metrics <- function(model_id, adjusted, robust, fits = model_fits) {
  lev_type <- ifelse(adjusted, "adjusted", "unadjusted")
  est_method <- ifelse(robust, "robust", "ols")
  model_key <- paste0("model_", model_id)

  model_fit <- fits[[lev_type]][[est_method]][[model_key]]$fit

  if (!inherits(model_fit, c("lmrob", "lm"))) {
    stop("Unsupported model class: ", class(model_fit)[1])
  }

  fit_sum <- summary(model_fit)
  n_obs <- length(fit_sum$residuals)
  n_params <- length(coef(model_fit)) + 1

  # compute information criteria (built-in for lm, approximated for lmrob)
  if (inherits(model_fit, "lm") && !inherits(model_fit, "lmrob")) {
    log_lik <- logLik(model_fit)
    aic <- AIC(model_fit)
    bic <- BIC(model_fit)
    rse_lbl <- "Residual_Std_Error"
  } else {
    # approximate log-likelihood for robust models assuming normal errors
    log_lik <- -n_obs / 2 * (log(2 * pi) + 2 * log(fit_sum$sigma) + 1)
    aic <- -2 * log_lik + 2 * n_params
    bic <- -2 * log_lik + log(n_obs) * n_params
    rse_lbl <- "Robust_RSE"
  }

  metrics <- data.frame(
    R_squared = fit_sum$r.squared,
    Adj_R_squared = fit_sum$adj.r.squared,
    RSE = fit_sum$sigma,
    AIC = aic,
    BIC = bic
  )
  colnames(metrics)[3] <- rse_lbl

  metrics
}

4 Nested Model Comparisons

Code 
# compare two nested models using Wald and deviance tests
compare_models <- function(mod1, mod2) {
  extract_p_value <- function(x) {
    pr_cols <- grep("^Pr", names(x), value = TRUE)
    if (length(pr_cols) == 0) {
      return(NA_real_)
    }
    tail(x[[pr_cols[1]]], 1)
  }

  tryCatch(
    {
      if (inherits(mod1, "lmrob")) {
        wald <- anova(mod1, mod2, test = "Wald")
        dev <- anova(mod1, mod2, test = "Deviance")

        data.frame(
          df1 = mod1$degree.freedom,
          df2 = mod2$degree.freedom,
          wald_stat = tail(wald$Test.Stat, 1),
          wald_p = extract_p_value(wald),
          deviance_stat = tail(dev$Test.Stat, 1),
          deviance_p = extract_p_value(dev),
          df = tail(wald$Df, 1)
        )
      } else {
        wald <- lmtest::waldtest(mod1, mod2, test = "Chisq")
        aov <- anova(mod1, mod2)

        data.frame(
          df1 = mod1$df.residual,
          df2 = mod2$df.residual,
          wald_stat = wald$Chisq[2],
          wald_p = extract_p_value(wald),
          deviance_stat = aov$F[2],
          deviance_p = extract_p_value(aov),
          df = aov$Df[2]
        )
      }
    },
    error = function(e) {
      warning("Model comparison failed: ", e$message)
      data.frame(
        df1 = NA, df2 = NA, wald_stat = NA, wald_p = NA,
        deviance_stat = NA, deviance_p = NA, df = NA
      )
    }
  )
}

# generate nested model comparison table
compare_nested_models <- function(fits, leverage, method,
                                  col_names = NULL, footnote_text = NULL) {
  models <- purrr::map(fits[[leverage]][[method]], ~ .x$fit)

  mn <- names(models)
  model_nums <- as.integer(sub(".*_", "", mn))

  ord <- order(model_nums)
  models <- models[ord]
  mn <- mn[ord]
  model_nums <- model_nums[ord]

  is_nested <- function(mod1, mod2) {
    terms1 <- attr(terms(mod1), "term.labels")
    terms2 <- attr(terms(mod2), "term.labels")
    all(terms1 %in% terms2) && !setequal(terms1, terms2)
  }

  num_combs <- combn(model_nums, 2, simplify = FALSE)
  name_combs <- combn(mn, 2, simplify = FALSE)

  nested_idx <- purrr::map_lgl(name_combs, function(nv) {
    is_nested(models[[nv[1]]], models[[nv[2]]])
  })

  num_combs <- num_combs[nested_idx]
  name_combs <- name_combs[nested_idx]

  model_pairs <- Map(
    function(nums, nv) setNames(nums, nv),
    num_combs, name_combs
  )

  comps <- model_pairs %>%
    purrr::map_df(~ {
      mod1 <- models[[names(.x)[1]]]
      mod2 <- models[[names(.x)[2]]]
      data.frame(
        model_a = .x[1],
        model_b = .x[2],
        model_pair = sprintf("%d vs. %d", .x[1], .x[2]),
        suppress_output(compare_models(mod1, mod2))
      )
    })

  labs <- glue("Model {comps$model_a}")

  comps %>%
    dplyr::rowwise() %>%
    dplyr::mutate(
      wald_stat = format_coef(wald_stat, wald_p),
      deviance_stat = format_coef(deviance_stat, deviance_p)
    ) %>%
    dplyr::ungroup() %>%
    dplyr::select(-model_a, -model_b, -wald_p, -deviance_p) %>%
    as_kable() %>%
    kableExtra::pack_rows(index = row_grouping(labs), escape = FALSE) %>%
    footnote() %>%
    add_style()
}
