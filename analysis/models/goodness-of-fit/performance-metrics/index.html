Purpose Validate model performance and stability using bootstrap-derived fit metrics
Inputs adjusted_data.rds, unadjusted_fits.rds, adjusted_fits.rds
Outputs gof_metrics.RData, gof_model_<k>.RData, summary_table_non_bootstrapped.html, summary_table_bootstrapped.html
Method MM-estimation robust regression; bootstrap resampling; goodness-of-fit metrics (C-index, calibration, weights, errors, R², RMSE, AIC/BIC)
Requires Step 7 (Unadjusted Models), Step 8 (Influence Diagnostics), Step 9 (Adjusted Models)
NoteRationale
  • MM-estimation: combines high breakdown point with efficiency under normal errors (Yohai 1987)
  • Robust linear regression: sharpened MM-estimator with adaptive scale (Koller and Stahel 2011)
  • Bootstrap intervals: nonparametric quantification of sampling variability (Efron and Tibshirani 1993)
  • R² resampling stability: indicates fit consistency across bootstrap samples (Davison and Hinkley 1997)

c_index.default <- function(boot_ests, y, x, n_cores = parallel::detectCores() - 1, ...) {

# model method: extracts components from model object
c_index.model <- function(obj, n_cores = parallel::detectCores() - 1, ...) {
  if (!all(c("bootstrap", "fit") %in% names(obj))) {
    stop("Model must contain 'bootstrap' and 'fit'")
  }
  bootstrap_stats <- .c_index_core(
    obj$bootstrap$ests,
    obj$fit$model[[1]],
    obj$fit$x,
    n_cores
  )
  list(bootstrap = bootstrap_stats)
}

# input validation helper
.validate_c_index <- function(boot_ests, y, x) {
  if (!is.matrix(boot_ests)) stop("boot_ests must be a matrix")
  if (!is.numeric(y)) stop("y must be numeric")
  if (!is.matrix(x)) stop("x must be a matrix")
  if (length(y) != nrow(x)) stop("Dimension mismatch: length(y) != nrow(x)")
}

# core computation: c-index across bootstrap samples
.c_index_core <- function(boot_ests, y, x, n_cores) {
  .validate_c_index(boot_ests, y, x)
  n_cores <- max(1, min(n_cores, parallel::detectCores() - 1))

  # computes concordance for a single coefficient vector
  calc_c_statistic <- function(coef) {
    predictions <- x %*% coef
    n <- length(y)
    chunk_size <- 1000L
    concordant_sum <- 0
    tied_sum <- 0
    total_pairs <- 0

    # compare pairs in blocks to manage memory
    for (i in seq(1, n, by = chunk_size)) {
      end_i <- min(i + chunk_size - 1, n)
      for (j in seq(i, n, by = chunk_size)) {
        end_j <- min(j + chunk_size - 1, n)
        obs_diff <- outer(y[i:end_i], y[j:end_j], "-")
        pred_diff <- outer(predictions[i:end_i], predictions[j:end_j], "-")
        keep <- if (i == j) upper.tri(obs_diff) else array(TRUE, dim(obs_diff))
        obs_diff_upper <- obs_diff[keep]
        pred_diff_upper <- pred_diff[keep]
        concordant_pairs <- obs_diff_upper != 0
        concordant_sum <- concordant_sum +
          sum(sign(obs_diff_upper[concordant_pairs]) == sign(pred_diff_upper[concordant_pairs]))
        tied_sum <- tied_sum + sum(pred_diff_upper[concordant_pairs] == 0)
        total_pairs <- total_pairs + sum(concordant_pairs)
      }
    }

    if (total_pairs == 0) {
      return(NA_real_)
    }
    (concordant_sum + 0.5 * tied_sum) / total_pairs
  }

  # windows-compatible parallel execution via clusters
  parallel_process <- function() {
    cluster <- parallel::makeCluster(n_cores)
    on.exit(parallel::stopCluster(cluster))
    parallel::clusterExport(
      cluster,
      c("boot_ests", "y", "x", "calc_c_statistic"),
      envir = environment()
    )
    pbapply::pbsapply(
      seq_len(nrow(boot_ests)),
      function(i) calc_c_statistic(boot_ests[i, ]),
      cl = cluster
    )
  }

  # os-specific parallel dispatch
  c_values <- if (.Platform$OS.type == "windows") {
    tryCatch(parallel_process(), error = function(e) stop("Parallel error: ", e$message))
  } else {
    if (!requireNamespace("pbmcapply", quietly = TRUE)) {
      stop("Package 'pbmcapply' is required for parallel processing.")
    }
    unlist(pbmcapply::pbmclapply(
      seq_len(nrow(boot_ests)),
      function(i) calc_c_statistic(boot_ests[i, ]),
      mc.cores = n_cores
    ))
  }

  # return summary statistics
  list(
    mean = mean(c_values, na.rm = TRUE),
    sd = sd(c_values, na.rm = TRUE),
    ci = stats::quantile(c_values, c(0.025, 0.975), na.rm = TRUE),
    values = c_values
  )
}

# input validation helper
.validate_calibration_data <- function(boot_coefs, obs_vals, x) {
  stopifnot(
    is.matrix(boot_coefs),
    is.numeric(obs_vals),
    is.matrix(x),
    length(obs_vals) == nrow(x),
    !anyNA(c(boot_coefs, obs_vals, x))
  )
}

# core computation: calibration metrics across bootstrap samples
.calibration_core <- function(boot_coefs, obs_vals, x) {
  .validate_calibration_data(boot_coefs, obs_vals, x)

  compute_stats <- function(coef) {
    predictions <- x %*% coef
    fit <- lm(obs_vals ~ predictions)
    coefs <- coef(fit)
    c(
      slope = coefs[2],
      intercept = coefs[1],
      r2 = summary(fit)$r.squared,
      citl = mean(obs_vals - predictions),
      e_max = max(abs(predictions - obs_vals))
    )
  }

  stats_matrix <- t(apply(boot_coefs, 1, compute_stats))
  colnames(stats_matrix) <- c("slope", "intercept", "r2", "citl", "e_max")

  summary_df <- apply(stats_matrix, 2, function(x) {
    c(
      mean = mean(x, na.rm = TRUE),
      sd = sd(x, na.rm = TRUE),
      stats::quantile(x, c(0.025, 0.5, 0.975), na.rm = TRUE)
    )
  })
  as.data.frame(summary_df)
}

# model method: compute calibration for original fit and bootstrap samples
calibration_stats.model <- function(obj, ...) {
  predictions <- obj$fit$fitted.values
  obs_vals <- obj$fit$model[[1]]
  fit <- lm(obs_vals ~ predictions)
  coefs <- coef(fit)

  original_stats <- list(
    slope = coefs[2],
    intercept = coefs[1],
    r2 = summary(fit)$r.squared,
    citl = mean(obs_vals - predictions),
    e_max_fitted = max(abs(predictions - obs_vals))
  )

  bootstrap_stats <- .calibration_core(obj$bootstrap$ests, obs_vals, obj$fit$x)

  list(original = original_stats, bootstrap = bootstrap_stats)
}

# model method: extract and summarize robust weights
wt_metrics.model <- function(obj, boot_samples = 1000, ...) {
  robust_weights <- obj$fit$rweights
  stopifnot(is.numeric(robust_weights))

  # compute histogram (Freedman-Diaconis breaks) and density
  hist_obj <- hist(robust_weights, plot = FALSE, breaks = "FD")
  density_obj <- density(robust_weights)

  # bootstrap confidence interval for the mean
  boot_means <- replicate(boot_samples, mean(sample(robust_weights, replace = TRUE)))
  mean_ci <- stats::quantile(boot_means, c(0.025, 0.975))

  stats <- list(
    n = length(robust_weights),
    mean = mean(robust_weights),
    median = median(robust_weights),
    sd = sd(robust_weights),
    min = min(robust_weights),
    max = max(robust_weights),
    mean_ci = mean_ci,
    prop_zero = mean(robust_weights == 0),
    prop_small = mean(robust_weights < 0.1)
  )

  structure(
    list(
      values = robust_weights,
      histogram = hist_obj,
      density = density_obj,
      stats = stats
    ),
    class = "wt_metrics"
  )
}

# input validation
.validate_boot_error_input <- function(true_coefs, boot_ests) {
  if (!is.matrix(boot_ests) && !is.data.frame(boot_ests)) {
    stop("boot_ests must be a matrix or data.frame")
  }
  if (is.null(colnames(boot_ests)) || is.null(names(true_coefs))) {
    stop("Column names for boot_ests and names for true_coefs are required")

# core computation
.boot_error_core <- function(true_coefs, boot_ests) {
  stopifnot(ncol(boot_ests) == length(true_coefs))

  errors <- abs(sweep(boot_ests, 2, true_coefs, "-"))
  percent_errors <- 100 * sweep(errors, 2, abs(true_coefs), "/")

  data.frame(
    coefficient = names(true_coefs),
    mae = colMeans(errors, na.rm = TRUE),
    medae = apply(errors, 2, median, na.rm = TRUE),
    mape = colMeans(percent_errors, na.rm = TRUE)
  )
}

boot_errors.numeric <- function(true_coefs, boot_ests, ...) {
  .boot_error_core(true_coefs, boot_ests, ...)
}

boot_errors.model <- function(obj, ...) {
  list(bootstrap = .boot_error_core(obj$fit$coef, obj$bootstrap$ests, ...))
}

# generic dispatch for R-squared metrics
boot_r2 <- function(obj, ...) {
  UseMethod("boot_r2")
}

boot_r2.default <- function(obj, ...) {
  stop("Unsupported type for boot_r2")
}

# core computation
.boot_r2_core <- function(observed_y, predicted_y_boot, num_predictors) {
  sse <- colSums((observed_y - predicted_y_boot)^2, na.rm = TRUE)
  tss <- sum((observed_y - mean(observed_y, na.rm = TRUE))^2, na.rm = TRUE)
  r2_values <- 1 - sse / tss
  n <- length(observed_y)
  adjusted_r2_values <- 1 - (1 - r2_values) * (n - 1) / (n - num_predictors)

  list(
    raw = data.frame(r2 = r2_values, adj_r2 = adjusted_r2_values),
    summary = data.frame(
      metric = c("R2", "Adj R2"),
      mean = c(mean(r2_values, na.rm = TRUE), mean(adjusted_r2_values, na.rm = TRUE)),
      sd = c(sd(r2_values, na.rm = TRUE), sd(adjusted_r2_values, na.rm = TRUE)),
      ci_lower = c(
        stats::quantile(r2_values, 0.025, na.rm = TRUE),
        stats::quantile(adjusted_r2_values, 0.025, na.rm = TRUE)
      ),
      ci_upper = c(
        stats::quantile(r2_values, 0.975, na.rm = TRUE),
        stats::quantile(adjusted_r2_values, 0.975, na.rm = TRUE)
      )
    )
  )
}

boot_r2.numeric <- function(y, y_hat, p, ...) {
  .boot_r2_core(y, y_hat, p)
}

boot_r2.model <- function(obj, ...) {
  observed_y <- obj$fit$model[[1]]
  predicted_y <- obj$fit$fitted.values
  sse <- sum((observed_y - predicted_y)^2)
  tss <- sum((observed_y - mean(observed_y))^2)
  r2 <- 1 - sse / tss
  n <- length(observed_y)
  p <- length(obj$fit$coef)
  adj_r2 <- 1 - (1 - r2) * (n - 1) / (n - p)

  original_stats <- list(r2 = r2, adj_r2 = adj_r2)

  predicted_y_boot <- obj$fit$x %*% t(obj$bootstrap$ests)
  bootstrap_stats <- .boot_r2_core(observed_y, predicted_y_boot, p)

  list(original = original_stats, bootstrap = bootstrap_stats)
}

# generic dispatch
boot_gof <- function(obj, ...) {
  UseMethod("boot_gof")
}

boot_gof.default <- function(obj, ...) {
  stop("Unsupported type for boot_gof")
}

.boot_gof_core <- function(boot_coefs, x_matrix, y_values, metrics = c("aic", "bic")) {
  predictions <- boot_coefs %*% t(x_matrix)
  residual_sum_squares <- rowSums((y_values - t(predictions))^2)
  n <- length(y_values)
  p <- ncol(boot_coefs)

  results <- data.frame(
    aic = n * log(residual_sum_squares / n) + 2 * p,
    bic = n * log(residual_sum_squares / n) + log(n) * p
  )[metrics]

  list(
    raw = results,
    summary = data.frame(
      metric = toupper(metrics),
      mean = colMeans(results, na.rm = TRUE),
      sd = sapply(results, sd, na.rm = TRUE)
    )
  )
}

boot_gof.direct <- function(boot_coefs, x, y, ...) {
  .boot_gof_core(boot_coefs, x, y, ...)
}

boot_gof.model <- function(obj, ...) {
  residuals <- obj$fit$model[[1]] - obj$fit$fitted.values
  rss <- sum(residuals^2)
  n <- length(obj$fit$model[[1]])
  p <- length(obj$fit$coef)

  original_stats <- list(
    aic = n * log(rss / n) + 2 * p,
    bic = n * log(rss / n) + log(n) * p
  )

  bootstrap_stats <- .boot_gof_core(obj$bootstrap$ests, obj$fit$x, obj$fit$model[[1]])

  list(original = original_stats, bootstrap = bootstrap_stats)
}

        # save per-model data for debugging

        model_idx <- as.integer(gsub("model_", "", model_name))

        metrics
      },
      error = function(e) {
        warning("Error processing ", model_name, ": ", conditionMessage(e))
        NULL
      }
    )
  })

  # persist aggregated metrics
  save(gof_metrics, file = gof_metrics_path)

# extract gof metrics from a single model fit
extract_gof <- function(model_fit, bootstrap = FALSE, digits = 3, linebreak = FALSE) {
  if (!is.list(model_fit)) stop("model_fit must be a list")
  required_keys <- c("boot_r2", "boot_gof")
  if (!all(required_keys %in% names(model_fit))) {
    stop("model_fit must contain 'boot_r2' and 'boot_gof'")
  }

  if (!bootstrap) {
    # validate original metrics structure
    if (!all(c("original") %in% names(model_fit$boot_r2)) ||
      !all(c("r2", "adj_r2") %in% names(model_fit$boot_r2$original))) {
      stop("model_fit$boot_r2 must contain 'original' with 'r2' and 'adj_r2'")
    }
    if (!all(c("original") %in% names(model_fit$boot_gof)) ||
      !all(c("aic", "bic") %in% names(model_fit$boot_gof$original))) {
      stop("model_fit$boot_gof must contain 'original' with 'aic' and 'bic'")
    }

    estimates <- c(
      R2 = model_fit$boot_r2$original$r2,
      Adj_R2 = model_fit$boot_r2$original$adj_r2,
      AIC = model_fit$boot_gof$original$aic,
      BIC = model_fit$boot_gof$original$bic
    )

    return(purrr::map(
      estimates, format_coef,
      pval = NA, se = NULL, digits = digits, linebreak = linebreak
    ))
  }

  # validate bootstrap metrics structure
  if (!all(c("bootstrap") %in% names(model_fit$boot_r2)) ||
    !all(c("summary") %in% names(model_fit$boot_r2$bootstrap)) ||
    !is.data.frame(model_fit$boot_r2$bootstrap$summary)) {
    stop("model_fit$boot_r2 must contain 'bootstrap$summary' as a data frame")
  }
  if (!all(c("bootstrap") %in% names(model_fit$boot_gof)) ||
    !all(c("summary", "raw") %in% names(model_fit$boot_gof$bootstrap)) ||
    !is.data.frame(model_fit$boot_gof$bootstrap$summary)) {
    stop("model_fit$boot_gof must contain 'bootstrap$summary' and 'bootstrap$raw'")
  }

  r2_summary <- model_fit$boot_r2$bootstrap$summary
  gof_summary <- model_fit$boot_gof$bootstrap$summary

  # validate required metrics present
  required_metrics <- list(r2_summary = c("R2", "Adj R2"), gof_summary = c("AIC", "BIC"))
  purrr::walk2(
    required_metrics, list(r2_summary, gof_summary),
    ~ if (!all(.x %in% .y$metric)) stop(paste(.x, "must be present"))
  )

  means <- c(
    R2 = r2_summary$mean[r2_summary$metric == "R2"],
    Adj_R2 = r2_summary$mean[r2_summary$metric == "Adj R2"]
  )
  sds <- c(
    R2 = r2_summary$sd[r2_summary$metric == "R2"],
    Adj_R2 = r2_summary$sd[r2_summary$metric == "Adj R2"]
  )
  r2_cells <- purrr::map2(
    means, sds, format_coef,
    pval = NA, digits = digits, linebreak = linebreak
  )

  # aic/bic: heavy-tailed bootstrap draws make mean/sd uninformative, so
  # report the median with the 2.5-97.5 percentile interval of raw draws
  raw_gof <- model_fit$boot_gof$bootstrap$raw
  fmt_median_ci <- function(x) {
    x <- x[is.finite(x)]
    if (length(x) == 0) {
      return(format_coef(
        NA_real_,
        pval = NA, se = NULL, digits = digits, linebreak = linebreak
      ))
    }
    q <- stats::quantile(x, c(0.5, 0.025, 0.975))
    sprintf(
      paste0("%.", digits, "f [%.", digits, "f, %.", digits, "f]"),
      q[1], q[2], q[3]
    )
  }
  gof_cells <- list(
    AIC = fmt_median_ci(raw_gof$aic),
    BIC = fmt_median_ci(raw_gof$bic)
  )

  c(r2_cells, gof_cells)
}

# combine gof metrics for summary table across models
combine_gof_metrics_summary <- function(gof) {
  # detect null model
  is_null_model <- !is.null(gof$c_index$bootstrap$values) &&
    all(is.na(gof$c_index$bootstrap$values))

  # c-index
  cindex_df <- if (!is.null(gof$c_index$bootstrap$mean)) {
    data.frame(
      Metric = "C-index",
      Value = gof$c_index$bootstrap$mean,
      SD = gof$c_index$bootstrap$sd,
      ci_lower = unname(gof$c_index$bootstrap$ci[1]),
      ci_upper = unname(gof$c_index$bootstrap$ci[2]),
      stringsAsFactors = FALSE
    )
  } else {
    data.frame(
      Metric = "C-index",
      Value = NA_real_, SD = NA_real_, ci_lower = NA_real_, ci_upper = NA_real_,
      stringsAsFactors = FALSE
    )
  }

  # calibration metrics
  calib_original <- gof$calibration_stats$original
  calib_boot <- gof$calibration_stats$bootstrap

  metrics_map <- list(
    slope = "Calibration Slope",
    intercept = "Calibration Intercept",
    r2 = "Calibration R²",
    citl = "Calibration CITL",
    e_max = "Calibration E-max"
  )

  get_orig_val <- function(key) {
    orig_key <- if (key == "e_max") "e_max_fitted" else key
    if (!is.null(calib_original[[orig_key]])) calib_original[[orig_key]] else NA_real_
  }

  # value = bootstrap mean, consistent with the per-model tables and the
  # bootstrap percentile intervals; falls back to the original-fit value
  cal_vals <- sapply(names(metrics_map), function(m) {
    if (!is.null(calib_boot) && m %in% colnames(calib_boot)) {
      as.numeric(calib_boot["mean", m])
    } else {
      get_orig_val(m)
    }
  })

  cal_ci_lower <- sapply(names(metrics_map), function(m) {
    if (!is.null(calib_boot) && m %in% colnames(calib_boot)) {
      as.numeric(calib_boot["2.5%", m])
    } else {
      NA_real_
    }
  })

  cal_ci_upper <- sapply(names(metrics_map), function(m) {
    if (!is.null(calib_boot) && m %in% colnames(calib_boot)) {
      as.numeric(calib_boot["97.5%", m])
    } else {
      NA_real_
    }
  })

  cal_dfs <- data.frame(
    Metric = unlist(metrics_map),
    Value = as.numeric(cal_vals),
    SD = NA_real_,
    ci_lower = as.numeric(cal_ci_lower),
    ci_upper = as.numeric(cal_ci_upper),
    stringsAsFactors = FALSE,
    row.names = NULL
  )

  # weight metrics
  wt_dfs <- if (!is.null(gof$wt_metrics)) {
    data.frame(
      Metric = c("Weight n", "Weight Distribution"),
      Value = c(gof$wt_metrics$stats$n, gof$wt_metrics$stats$mean),
      SD = c(NA, gof$wt_metrics$stats$sd),
      ci_lower = c(NA, gof$wt_metrics$stats$mean_ci["2.5%"]),
      ci_upper = c(NA, gof$wt_metrics$stats$mean_ci["97.5%"]),
      stringsAsFactors = FALSE
    )
  } else {
    data.frame(
      Metric = c("Weight n", "Weight Distribution"),
      Value = NA_real_, SD = NA_real_, ci_lower = NA_real_, ci_upper = NA_real_,
      stringsAsFactors = FALSE
    )
  }

  # null model adjustments
  if (is_null_model) {
    cindex_df[c("Value", "SD", "ci_lower", "ci_upper")] <- NA
    blank_metrics <- c("Calibration Slope", "Calibration R²", "Calibration E-max", "Calibration Intercept")
    cal_dfs[cal_dfs$Metric %in% blank_metrics, c("Value", "ci_lower", "ci_upper")] <- NA
  }

  combined_df <- dplyr::bind_rows(cindex_df, cal_dfs, wt_dfs)

  # format CI and final output
  combined_df <- combined_df %>%
    dplyr::mutate(
      CI = purrr::map2_chr(ci_lower, ci_upper, function(l, u) {
        if (is.na(l) || is.na(u)) {
          return(NA_character_)
        }
        sprintf("%.3f–%.3f", l, u)
      })
    ) %>%
    dplyr::mutate(
      Formatted = purrr::pmap_chr(
        list(Value, SD, CI),
        function(v, s, c) {
          if (exists("format_gof_html")) {
            format_gof_html(v, s, c, digs = 3)
          } else {
            val_str <- if (is.na(v)) "–" else sprintf("%.3f", v)
            ci_str <- if (is.na(c)) "" else paste0(" (", c, ")")
            paste0(val_str, ci_str)
          }
        }
      )
    ) %>%
    dplyr::select(Metric, Formatted)

  combined_df
}

# format gof value with optional sd and ci for html display
format_gof_html <- function(val, sd = NA, ci = NA, digs = 3) {
  fmt_num <- function(num, d) sprintf("%.*f", d, round(num, d))

  if (is.na(val)) {
    return("–")
  }

  val_str <- paste0(
    if (val >= 0 && val < 10) "<span style='display:inline-block;width:4px'></span>" else "",
    fmt_num(val, digs)
  )
  sd_str <- if (!is.na(sd)) {
    sprintf("<br><span style='font-size:0.8em'>(±%s)</span>", fmt_num(sd, digs))
  } else {
    ""
  }
  ci_str <- if (!is.na(ci)) {
    sprintf("<br><span style='font-size:0.8em; white-space: nowrap;'>[%s]</span>", ci)
  } else {
    ""
  }

  paste0(val_str, sd_str, ci_str)
}

# combine detailed gof metrics for per-model display
combine_gof_metrics <- function(gof) {
  # c-index
  cindex <- data.frame(
    component = "c_index",
    metric = "c_index",
    estimate = gof$c_index$bootstrap$mean,
    sd = gof$c_index$bootstrap$sd,
    ci_lower = unname(gof$c_index$bootstrap$ci["2.5%"]),
    ci_upper = unname(gof$c_index$bootstrap$ci["97.5%"])
  )

  # calibration
  calib_stats <- gof$calibration_stats$bootstrap
  calib <- data.frame(
    component = "calibration",
    metric = colnames(calib_stats),
    estimate = as.numeric(calib_stats["mean", ]),
    sd = as.numeric(calib_stats["sd", ]),
    ci_lower = as.numeric(calib_stats["2.5%", ]),
    ci_upper = as.numeric(calib_stats["97.5%", ])
  )

  # weight metrics
  wt <- data.frame(
    component = "wt_metrics",
    metric = c("mean", "median", "min", "max"),
    estimate = c(
      gof$wt_metrics$stats$mean,
      gof$wt_metrics$stats$median,
      gof$wt_metrics$stats$min,
      gof$wt_metrics$stats$max
    ),
    sd = c(gof$wt_metrics$stats$sd, NA, NA, NA),
    ci_lower = c(gof$wt_metrics$stats$mean_ci["2.5%"], NA, NA, NA),
    ci_upper = c(gof$wt_metrics$stats$mean_ci["97.5%"], NA, NA, NA)
  )

  # bootstrap errors
  boot_err <- gof$boot_errors$bootstrap %>%
    dplyr::summarise(mae = mean(mae), medae = mean(medae), mape = mean(mape)) %>%
    tidyr::pivot_longer(
      cols = dplyr::everything(),
      names_to = "metric",
      values_to = "estimate"
    ) %>%
    dplyr::mutate(component = "boot_errors", sd = NA, ci_lower = NA, ci_upper = NA) %>%
    dplyr::select(component, metric, estimate, sd, ci_lower, ci_upper)

  # bootstrap r2
  boot_r2 <- gof$boot_r2$bootstrap$summary %>%
    dplyr::rename(estimate = mean) %>%
    dplyr::mutate(component = "boot_r2") %>%
    dplyr::select(component, metric, estimate, sd, ci_lower, ci_upper)

  # bootstrap rmse
  boot_rmse <- data.frame(
    component = "boot_rmse",
    metric = "rmse",
    estimate = gof$boot_rmse$bootstrap$mean_coef_rmse,
    sd = gof$boot_rmse$bootstrap$sd_coef_rmse,
    ci_lower = NA,
    ci_upper = NA
  )

  # bootstrap gof (aic, bic): heavy-tailed draws make mean/sd uninformative,
  # so report the median with the 2.5-97.5 percentile interval of raw draws
  med_or_na <- function(x) {
    x <- x[is.finite(x)]
    if (length(x) == 0) NA_real_ else stats::median(x)
  }
  q_or_na <- function(x, p) {
    x <- x[is.finite(x)]
    if (length(x) == 0) NA_real_ else as.numeric(stats::quantile(x, p))
  }
  gof_raw <- gof$boot_gof$bootstrap$raw
  boot_gof <- data.frame(
    component = "boot_gof",
    metric = c("AIC", "BIC"),
    estimate = c(med_or_na(gof_raw$aic), med_or_na(gof_raw$bic)),
    sd = NA_real_,
    ci_lower = c(q_or_na(gof_raw$aic, 0.025), q_or_na(gof_raw$bic, 0.025)),
    ci_upper = c(q_or_na(gof_raw$aic, 0.975), q_or_na(gof_raw$bic, 0.975))
  )

  # combine all
  tbl <- dplyr::bind_rows(cindex, calib, wt, boot_err, boot_r2, boot_rmse, boot_gof)

  # weight distribution histogram
  hist_plot <- if (inherits(gof$wt_metrics, "wt_metrics") &&
    all(c("values", "histogram", "density") %in% names(gof$wt_metrics))) {
    plot(gof$wt_metrics) + garamond_theme
  } else {
    warning("Cannot plot wt_metrics: missing components or incorrect class.")
    NULL
  }

  list(summary = tbl, component = hist_plot)
}

# generate formatted gof table and figure for a single model
print_gof <- function(model) {
  gof_data <- combine_gof_metrics(model)

  # row groups for table organization
  row_groups <- beautify_names(gof_data$summary$component) %>%
    md_to_html() %>%
    row_grouping()

  names(row_groups) <- c(
    "C-Index", "Calibration", "Weighted Metrics", "Error Metrics",
    "Explained Variance", "RMSE", "Model Fit"
  )

  # metric labels
  metric_labels <- list(
    c("C-Index"),
    c("Slope", "Intercept", "$R^2$", "CITL", "$E_{\\text{max}}$"),
    c("Mean", "Median", "Min", "Max"),
    c("MAE", "MedAE", "MAPE"),
    c("$R^2$", "Adj. $R^2$"),
    c("RMSE"),
    c("AIC", "BIC")
  )

  gof_table <- gof_data$summary %>%
    dplyr::select(-component) %>%
    dplyr::mutate(metric = md_to_html(unlist(metric_labels))) %>%
    as_kable(sci_threshold = FALSE, digits = 3) %>%
    kableExtra::pack_rows(index = row_groups, escape = FALSE) %>%
    footnote() %>%
    add_style()

  list(table = gof_table, figure = gof_data$component)
}
