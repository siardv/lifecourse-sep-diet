Random Forest Approach via missForest

Author

Siard van den Bosch

Published

November 20, 2024

Modified

June 12, 2026

Purpose Impute missing values via random-forest imputation on partnered cases
Inputs data_without_outliers.sav
Outputs imputed_data_combined.sav, imputed_results_missforest.rds
Method missForest random-forest imputation; variablewise OOB error
Requires Step 4 (Outliers Detection)
NoteRationale
  • missForest: handles mixed numeric and factor variables without distributional assumptions (Stekhoven and Bühlmann 2012)
  • Partition by partnership: partner variables have structural NA for non-partnered cases and must not be imputed (Little and Rubin 2019)
  • variablewise = TRUE: yields per-variable OOB error for targeted diagnostics; imputed values are unaffected
  • Serial execution (parallelize = "no"): preserves variablewise OOB reporting; parallelize = "variables" returns zero-valued OOB in current missForest
  • OOB error: bootstrap aggregation in random forests provides internal validation (Breiman 2001)
  • NRMSE and PFC: standard fidelity metrics for mixed-type random-forest imputation (Tang and Ishwaran 2017)

4 Run missForest Imputation

Code 
# set seed for reproducibility
set.seed(imputation_params$seed)

# choose a safe number of workers (reserve one core for system)
n_workers <- max(1L, parallel::detectCores(logical = FALSE) - 1L)

# determine parallelization strategy based on problem size
missing_rate <- vapply(df_impute, function(col) mean(is.na(col)), numeric(1))
n_miss_vars <- sum(missing_rate > 0)

# IMPORTANT: parallelize = "variables" together with variablewise = TRUE
# returns a length-correct but all-zero OOBerror vector in current missForest
# (the imputation itself completes correctly but OOB diagnostics are useless).
# Serial execution preserves per-variable OOB reporting. Runtime impact is
# modest for this dataset (911 rows, ~31 variables, 200 trees): tens of
# seconds to about a minute on a modern laptop.
parallelize <- "no"

# detect missForest capabilities (legacy randomForest vs ranger backend)
mf_formals <- names(formals(missForest::missForest))
has_backend <- "backend" %in% mf_formals
has_num_threads <- "num.threads" %in% mf_formals

# do not pass a backend argument; let missForest use its default.
# Earlier attempts to explicitly set backend = "ranger" or to suppress it in
# combination with parallelize = "variables" did not restore OOB reporting;
# the serial parallelize = "no" path above is what fixes it.
backend <- NULL

# set up foreach backend only when missForest needs it
use_foreach <- parallelize == "variables" ||
  (parallelize == "forests" && !has_num_threads)

if (use_foreach) {
  clust <- parallel::makeCluster(n_workers)
  on.exit(parallel::stopCluster(clust), add = TRUE)
  doParallel::registerDoParallel(clust)
  doRNG::registerDoRNG(seed = imputation_params$seed)
}

# assemble missForest arguments
mf_args <- list(
  xmis = df_impute,
  ntree = imputation_params$ntree,
  mtry = imputation_params$mtry,
  maxiter = imputation_params$maxiter,
  variablewise = imputation_params$variablewise,
  verbose = TRUE,
  parallelize = parallelize,
  replace = imputation_params$replace
)

if (has_num_threads && parallelize == "forests") mf_args$num.threads <- n_workers

# execute imputation
imp_mf <- do.call(missForest::missForest, mf_args)

# create base dataframe for variable-wise metrics
mf_metrics <- data.frame(
  var = colnames(df_impute),
  erm = ifelse(vapply(df_impute, is.numeric, logical(1)), "NRMSE (approx)", "PFC"),
  oob = as.numeric(raw_oob),
  stringsAsFactors = FALSE
) %>%
  dplyr::left_join(lookup_sd, by = "var") %>%
  dplyr::left_join(missing_info, by = "var") %>%
  dplyr::mutate(
    # for numeric variables, convert OOB MSE to approximate NRMSE
    oob = dplyr::case_when(
      erm == "NRMSE (approx)" & is.finite(sd_obs) & sd_obs > 0 ~ sqrt(oob) / sd_obs,
      erm == "NRMSE (approx)" ~ NA_real_,
      TRUE ~ oob
    ),
    # add a compact rating to prioritize review
    rating = dplyr::case_when(
      missing_rate == 0 ~ "Complete",
      oob < 0.5 ~ "Excellent",
      oob < 0.7 ~ "Good",
      oob < 0.9 ~ "Acceptable",
      TRUE ~ "Review"
    )
  ) %>%
  dplyr::select(var, erm, oob, missing_rate, rating)

# compute a global, missingness-weighted summary from variable-wise metrics
global_df <- mf_metrics %>%
  dplyr::filter(missing_rate > 0) %>%
  dplyr::group_by(erm) %>%

    rating = "Global (weighted)",

  dplyr::mutate(var = "Global") %>%

# function to create density comparison plots for imputation diagnostics
plot_imputation_diagnostic <- function(orig_data, imputed_data, variable) {
  x_orig <- orig_data[[variable]]
  x_imp <- imputed_data[[variable]]

  miss_idx <- is.na(x_orig)
  n_miss <- sum(miss_idx)
  n_tot <- length(x_orig)

  # nothing to diagnose if no missingness
  if (n_miss == 0) {
    return(invisible(NULL))
  }

  # caption text showing missing rate
  caption_txt <- sprintf("Missing: %d (%.1f%%)", n_miss, 100 * n_miss / n_tot)

  # factor/character path: bar chart of proportions
  if (is.factor(x_orig) || is.character(x_orig)) {
    x_orig_f <- as.factor(x_orig)
    x_imp_f <- as.factor(x_imp)

    # observed distribution (excluding NAs)
    obs_tab <- prop.table(table(x_orig_f[!is.na(x_orig_f)]))

    # imputed-at-missing distribution
    imp_tab <- prop.table(table(x_imp_f[miss_idx]))

    levs <- union(names(obs_tab), names(imp_tab))
    df_plot <- data.frame(
      level = levs,
      observed = as.numeric(obs_tab[levs]),
      imputed_missing = as.numeric(imp_tab[levs])
    )
    df_plot[is.na(df_plot)] <- 0

    df_long <- df_plot %>%
      tidyr::pivot_longer(
        cols = c("observed", "imputed_missing"),
        names_to = "source",
        values_to = "prop"
      ) %>%
      dplyr::mutate(
        source = factor(
          source,
          levels = c("observed", "imputed_missing"),
          labels = c("Observed (non-missing)", "Imputed (missing only)")
        )
      )

    print(
      ggplot2::ggplot(df_long, ggplot2::aes(x = level, y = prop, fill = source)) +
        ggplot2::geom_col(position = "dodge") +
        ggplot2::labs(
          title = gsub("<.*?>", "", beautify_names(variable)),
          x = gsub("<.*?>", "", beautify_names(variable)),
          y = "Proportion",
          fill = NULL,
          caption = caption_txt
        ) +
        ggplot2::theme(axis.text.x = ggplot2::element_text(angle = 45, hjust = 1))
    )

    return(invisible(NULL))
  }

  # numeric path: density comparison
  x_orig_num <- as.numeric(x_orig)
  x_obs <- x_orig_num[!miss_idx]
  x_imp_miss <- as.numeric(x_imp)[miss_idx]

  # keep only finite values
  x_obs <- x_obs[is.finite(x_obs)]
  x_imp_miss <- x_imp_miss[is.finite(x_imp_miss)]

  # if too few points for density, fall back to point plot
  if (length(x_imp_miss) < 2L || length(x_obs) < 2L) {
    df_fallback <- data.frame(
      value = c(x_obs, x_imp_miss),
      source = c(
        rep("Observed (non-missing)", length(x_obs)),
        rep("Imputed (missing only)", length(x_imp_miss))
      )
    )

    print(
      ggplot2::ggplot(df_fallback, ggplot2::aes(x = value, y = source)) +
        ggplot2::geom_point(alpha = 0.6) +
        ggplot2::labs(
          title = gsub("<.*?>", "", beautify_names(variable)),
          x = gsub("<.*?>", "", beautify_names(variable)),
          y = NULL,
          caption = paste0(caption_txt, " | density skipped (too few points)")
        )
    )

    return(invisible(NULL))
  }

  # compute kernel density estimates
  dens_obs <- stats::density(x_obs)
  dens_imp <- stats::density(x_imp_miss)

  df_dens <- data.frame(
    x = dens_obs$x,
    observed = dens_obs$y,
    imputed_missing = stats::approx(
      x = dens_imp$x, y = dens_imp$y,
      xout = dens_obs$x, rule = 2
    )$y
  )

  df_long <- df_dens %>%
    tidyr::pivot_longer(
      cols = c("observed", "imputed_missing"),
      names_to = "source",
      values_to = "density"
    ) %>%
    dplyr::mutate(
      source = factor(
        source,
        levels = c("observed", "imputed_missing"),
        labels = c("Observed (non-missing)", "Imputed (missing only)")
      )
    )

  print(
    ggplot2::ggplot(df_long, ggplot2::aes(x = x, y = density, color = source)) +
      ggplot2::geom_line(linewidth = 1.2) +
      ggplot2::labs(
        title = gsub("<.*?>", "", beautify_names(variable)),
        x = gsub("<.*?>", "", beautify_names(variable)),
        y = "Density",
        color = NULL,
        caption = caption_txt
      )
  )

  invisible(NULL)
}
