4 Define Classification Mapping Functions

Three functions: brc14_map() retrieves the CBS crosswalk, isco08_to_brc14() returns BRC14 group codes, and isco08_to_isei08() returns ISEI-08 scores.

Code 
# fetch BRC2014 mapping from local file or download if absent
brc14_map <- function() {
  local_file <- here::here("assets", "data", "brc2014.xls")
  if (!file.exists(local_file)) {
    download_url <- glue(
      "https://www.cbs.nl/-/media/imported/",
      "onze-diensten/methoden/classificaties/",
      "documents/2015/09/brc2014.xls"
    )
    download.file(download_url, local_file, mode = "wb")
  }
  m <- readxl::read_excel(local_file, sheet = 2)

  # code columns are zero-padded text in the workbook; parse to numeric so
  # armed-forces codes (e.g. '0110') match numeric isco08 values
  m$ISCO2008unitgroup <- as.numeric(m$ISCO2008unitgroup)
  m$BRC2014beroepsgroep <- as.numeric(m$BRC2014beroepsgroep)
  stopifnot(!anyNA(m$ISCO2008unitgroup), !anyNA(m$BRC2014beroepsgroep))
  m
}

# convert ISCO-08 codes to BRC14 classifications
isco08_to_brc14 <- function(x) {
  m <- brc14_map()
  purrr::map_dbl(x, ~ m$BRC2014beroepsgroep[
    match(.x, m$ISCO2008unitgroup, nomatch = NA)
  ])
}

# convert ISCO-08 codes to ISEI-08 scores with error handling
isco08_to_isei08 <- function(x) {
  sapply(x, function(code) {
    if (is.na(code)) {
      return(NA)
    }
    tryCatch(
      ISCO08ConveRsions::isco08toisei08(
        formatC(as.integer(code), width = 4, flag = "0", format = "d")
      ),
      error = function(e) NA
    )
  })
}

rename_and_select <- function(data, suffix = "") {
  require(rlang, quietly = TRUE)
  suffix <- if (suffix == "") suffix else paste0("_", suffix)

  out <- data %>%
    dplyr::rename(
      !!paste0("isco08_old", suffix) := isco08_init,
      !!paste0("occupation", if (suffix == "") "" else suffix, "_cleaned") := occupation,
      !!paste0("isco08", suffix) := isco08,
      !!paste0("isco08_new", suffix) := isco08_new
    ) %>%
    dplyr::select(-source)

  names(out) <- gsub("[_]+", "_", names(out))
  return(out)
}

# scatter plot with regression line to visualize code corrections
scatter <- function(x_var, y_var, title, point_size = 2) {
  # filter complete cases for plotting
  df_clean <- main %>%
    dplyr::select(dplyr::all_of(c(x_var, y_var))) %>%
    dplyr::filter(complete.cases(.))

  # compute linear model for trend visualization
  fit <- lm(as.formula(paste(y_var, "~", x_var)), data = df_clean)

  # add residuals and predictions to dataset
  df_clean$resids <- residuals(fit)
  df_clean$y_hat <- predict(fit)
  df_clean$resids_trans <- sqrt(abs(df_clean$resids))

  # generate scatter plot with color-coded deviations
  df_clean %>%
    ggplot2::ggplot(ggplot2::aes(
      x = .data[[x_var]],
      y = .data[[y_var]],
      color = resids_trans
    )) +
    ggplot2::geom_smooth(
      method = "lm",
      se = FALSE,
      color = "black",
      linewidth = 0.8,
      formula = y ~ x
    ) +
    ggplot2::geom_jitter(
      size = point_size,
      alpha = 0.6,
      width = 0.2,
      height = 0.2
    ) +
    ggplot2::scale_color_viridis_c(option = "plasma", name = "Deviation") +
    ggplot2::labs(
      title = title,
      x = "ISCO-08 (Corrected)",
      y = "ISCO-08 (Original)"
    )
}

# split strings into characters and pad for alignment
split_and_pad <- function(x, max_len) {
  chars <- unlist(strsplit(as.character(x), ""))
  pad <- rep("_", max_len - nchar(x))
  c(chars, pad)
}

# highlight differing digits in corrected codes
highlight_diff_digits <- function(a, b) {
  # return b if either input is NA
  if (anyNA(c(a, b))) {
    return(as.character(b))
  }

  # determine maximum length for padding
  max_len <- max(nchar(a), nchar(b))

  # split and pad strings to uniform length
  a_digs <- split_and_pad(a, max_len)
  b_digs <- split_and_pad(b, max_len)

  # define colors for unchanged and changed digits
  clr <- viridis::plasma(20, 1, 0)[seq(4, 12, length.out = 2)]

  # format digits with HTML color coding
  formatted <- paste0(
    mapply(function(init, new) {
      sprintf(
        '<span style="color:%s;">%s</span>',
        ifelse(init == new, clr[1], clr[2]),
        new
      )
    }, a_digs, b_digs),
    collapse = ""
  )

  # return b if formatting fails
  if (is.na(formatted)) {
    return(as.character(b))
  }
  return(formatted)
}

# generate summary table of ISCO corrections with explicit source name parameter
summarize_isco_corrections <- function(x, source_name = NULL) {
  # use provided source name or infer from variable name
  source_name <- source_name %||% deparse(substitute(x))

  # calculate total rows and justified corrections
  tot_rows <- nrow(x)
  just_df <- dplyr::filter(x, !is.na(isco08_new))
  just_rows <- nrow(just_df)
  pct_just <- round(just_rows / tot_rows * 100)

  footnote_text <- glue(
    "Correction records: $N={just_rows}$ ({pct_just}% of the {tot_rows} cases in ",
    "{source_name}). Occupation titles remain in lowercase for consistency with ",
    "reported data. Revised ISCO-08 codes reflect updates from original ",
    "classifications, supported by a rationale for each change. Red digits ",
    "highlight code modifications, while unchanged digits appear in purple."
  )

  # format table with styling and footnote
  just_df %>%
    as_kable() %>%
    kableExtra::column_spec(seq(ncol(just_df)), extra_css = "vertical-align: top;") %>%
    footnote() %>%
    add_style()
}
