Transforming Variables for Consistency and Imputation Readiness

Author

Siard van den Bosch

Published

August 26, 2024

Modified

June 12, 2026

Purpose Recode variables, screen missingness, and construct SES composites for imputation
Inputs main_corrected.sav
Outputs data_for_imputation.sav
Method Variable recoding; MCAR diagnostic; psychometric scale evaluation
Requires Step 2 (ISCO-08 Standardization)
NoteRationale
  • Critical-missing exclusion: drop cases missing both DHD index and kilocalories rather than impute pivotal variables (Enders 2010)
  • Retention threshold: 20% per-case missingness cap balances sample retention against imputation quality (Schafer and Graham 2002)
  • Structural missingness: partner indicators set to NA for unpartnered respondents to separate from MAR (Little 1988)
  • Domain-specific SES: education and occupation index distinct mechanisms (cultural capital vs. material resources) (Galobardes et al. 2006)
  • Scale validity criteria: \(\alpha\), AVE, Fornell-Larcker, and HTMT follow established psychometric frameworks (Fornell and Larcker 1981)

# update BRC14 variable labels from mapping table
update_brc14 <- function(column_name, data, mapping) {
  values <- unique(stats::na.omit(data[[column_name]]))
  labels <- sapply(values, function(x) {
    unique(mapping$BRC2014beroepsgroep_label[
      as.numeric(mapping$BRC2014beroepsgroep) == x
    ])
  })
  values_named <- setNames(values, labels)
  for (i in seq_along(values_named)) {
    eval.parent(substitute(
      update_value_label(
        data,
        column_name,
        as.numeric(values_named[i]),
        names(values_named)[i]
      )
    ))
  }
}

# update ISCO-08 variable labels from job title lookup
update_isco08r <- function(column_name, data) {
  values <- unique(stats::na.omit(data[[column_name]]))
  labels <- sapply(values, function(x) {
    isco08_to_jobtitle(x)
  })
  values_named <- setNames(values, labels)
  for (i in seq_along(values_named)) {
    eval.parent(substitute(
      update_value_label(
        data,
        column_name,
        as.numeric(values_named[i]),
        names(values_named)[i]
      )
    ))
  }
}

non_deterministic_data <- data_cleaned %>%

  # perform MCAR test with error handling

}

# helper function to summarize missingness distribution
summarize_missingness <- function(x) {
  counts <- as.numeric(x)
  props <- paste0(round(as.numeric(names(x)) * 100, 2), "%")
  parts <- paste0(counts[-1], " (", props[-1], ")")
  sentence <- glue(
    "Out of the total {sum(counts)} ",
    "respondents, {counts[1]} had no missing values, ",
    "followed by {paste(parts, collapse = ', ')} ",
    "with increasing proportions of missing values."
  )
  htmltools::HTML(as.character(htmltools::p(sentence)))
}

# function to compute descriptive statistics for SES indicators
compute_ses_descriptives <- function(data) {
  data %>%
    tidyr::pivot_longer(
      cols = dplyr::everything(),
      names_to = "indicator",
      values_to = "value"
    ) %>%
    dplyr::group_by(indicator) %>%
    dplyr::summarise(
      n = sum(!is.na(value)),
      n_miss = sum(is.na(value)),
      pct_miss = round(100 * mean(is.na(value)), 1),
      mean = round(mean(value, na.rm = TRUE), 2),
      sd = round(sd(value, na.rm = TRUE), 2),
      min = min(value, na.rm = TRUE),
      max = max(value, na.rm = TRUE),
      skew = round(moments::skewness(value, na.rm = TRUE), 2),
      kurt = round(moments::kurtosis(value, na.rm = TRUE), 2),
      .groups = "drop"
    ) %>%
    dplyr::mutate(
      domain = dplyr::case_when(
        grepl("edu", indicator) ~ "Education",
        grepl("occ", indicator) ~ "Occupation",
        TRUE ~ NA_character_
      )
    ) %>%
    dplyr::arrange(domain, indicator)
}

# function to compute average variance extracted (AVE)
compute_ave <- function(fit, factor_name) {
  loadings <- lavaan::standardizedSolution(fit) %>%
    dplyr::filter(op == "=~", lhs == factor_name) %>%
    dplyr::pull(est.std)
  mean(loadings^2)
}

# function to compute composite reliability (CR)
compute_cr <- function(fit, factor_name) {
  loadings <- lavaan::standardizedSolution(fit) %>%
    dplyr::filter(op == "=~", lhs == factor_name) %>%
    dplyr::pull(est.std)
  sum_l <- sum(loadings)
  sum_e <- sum(1 - loadings^2)
  (sum_l^2) / (sum_l^2 + sum_e)
}
