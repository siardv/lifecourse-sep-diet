Purpose Remove univariate and multivariate outliers from the analysis dataset
Inputs data_for_imputation.sav
Outputs data_without_outliers.sav, outlier_cases_removed.csv
Method IQR/MAD univariate detection; LOF multivariate detection
Requires Step 3 (Data Cleaning)
NoteRationale
  • IQR for dhd_kcal: classical robust rule for energy intake distributions (Tukey 1977)
  • MAD for dhd_index: robust to non-normality, 50% breakdown point, threshold 2.5 approximates 99% normal coverage (Leys et al. 2013)
  • LOF for multivariate detection: density-based, no distributional assumption (Breunig et al. 2000)
  • Tukey adjusted-boxplot cutoff: skew-corrected via medcouple, default alpha = 0.05 controls Type-I error (Hubert and Vandervieren 2008)
  • Partner variables excluded from complete-case filter: structural missingness for non-partnered respondents (Buuren 2018)
  • SES composites excluded from outlier detection: extreme socioeconomic values represent legitimate population variation, not measurement error (Aguinis et al. 2013)
  • Reconciled score and status columns (siops08*, egp11*, lf_status*) excluded from detection: preserves the complete-case pool and LOF feature set of the pre-reconciliation analysis

3 Detect Univariate Outliers

Code 
# detect univariate outliers using IQR, z-score, or MAD method
univariate_outliers <- function(data, var, meth = "iqr", thresh = NULL) {
  stopifnot(meth %in% c("iqr", "zscore", "mad"))

  # set method-specific default threshold
  if (is.null(thresh)) {
    thresh <- if (meth == "iqr") 1.5 else 2.5
  }

  # extract finite, non-NA values
  x <- data[[var]][!is.na(data[[var]]) & is.finite(data[[var]])]

  # calculate thresholds based on method
  lims <- switch(meth,
    "iqr" = {
      q <- quantile(x, c(0.25, 0.75), names = FALSE)
      iqr <- diff(q)
      q + c(-thresh, thresh) * iqr
    },
    "zscore" = mean(x) + c(-1, 1) * thresh * sd(x),
    "mad" = {
      med <- median(x)
      med + c(-1, 1) * thresh * mad(x)
    }
  )

  # identify outlier indices
  outs_idx <- which(x < lims[1] | x > lims[2])

  # compute mean and standard error
  stats_val <- c(mean = mean(x), se = sd(x) / sqrt(length(x)))

  # calculate 95% confidence interval
  ci <- stats_val["mean"] + c(-1.96, 1.96) * stats_val["se"]

  # return NULL if non-finite values present
  if (!all(is.finite(c(stats_val, ci)))) {
    return(NULL)
  }

  list(
    outliers = x[outs_idx],
    ids = data$idnummer[!is.na(data[[var]]) & is.finite(data[[var]])][outs_idx],
    method = meth,
    lower_threshold = lims[1],
    upper_threshold = lims[2],
    mean = stats_val["mean"],
    ci_lower = ci[1],
    ci_upper = ci[2]
  )
}

# detect outliers across multiple variables with specified methods
detect_outliers <- function(data, vars, meths) {
  data$outlier <- NA_character_

  # apply univariate detection to each variable
  res <- Map(function(v, m) univariate_outliers(data, v, m), vars, meths)

  # flag outliers in data
  for (i in seq_along(vars)) {
    if (!is.null(res[[i]])) {
      ids <- res[[i]]$ids
      label <- glue("Univariate ({beautify_names(vars[i])})")
      data$outlier[data$idnummer %in% ids] <- label
    }
  }

  # extract summary metrics
  metrics <- c(
    "method", "mean", "ci_lower", "ci_upper",
    "upper_threshold", "lower_threshold"
  )

  sum_data <- sapply(metrics, function(m) {
    v <- sapply(res, function(r) if (is.null(r)) NA else r[[m]])
    if (m == "method") toupper(as.character(v)) else round(as.numeric(v), 2)
  })

  # count detections
  detection_counts <- sapply(res, function(r) {
    if (is.null(r)) 0 else length(r$outliers)
  })

  # count unique cases (some may be flagged on multiple variables)
  all_outlier_ids <- unique(unlist(lapply(res, function(r) {
    if (!is.null(r)) r$ids else NULL
  })))
  unique_cases <- length(all_outlier_ids)

  list(
    summary = list(
      variable = c(vars, "Total detections", "Unique cases"),
      count = c(detection_counts, sum(detection_counts), unique_cases),
      tbl = rbind(sum_data, NA, NA)
    ),
    results = res,
    updated_data = data
  )
}

# create boxplot with outliers highlighted and confidence intervals
create_boxplot <- function(data, variable, ids, mean_val, ci_lower, ci_upper) {
  # define colors
  ci_color <- viridis::plasma(20)[4]
  outlier_color <- viridis::plasma(20)[12]
  data_color <- viridis::viridis(10)[4]

  mean_data <- data.frame(
    x = factor(1), y = mean_val,
    ymin = ci_lower, ymax = ci_upper
  )

  # set jitter position with seed for reproducibility
  pos_jitter <- if ("seed" %in% names(formals(ggplot2::position_jitter))) {
    ggplot2::position_jitter(width = 0.3, height = 0.25, seed = 42)
  } else {
    ggplot2::position_jitter(width = 0.3, height = 0.25)
  }

  ggplot2::ggplot(data, ggplot2::aes(x = factor(1), y = .data[[variable]])) +
    ggplot2::geom_boxplot(fill = "lightgrey", alpha = 0.5, outlier.shape = NA) +
    ggplot2::geom_point(
      data = dplyr::filter(data, idnummer %in% ids),
      ggplot2::aes(x = factor(1), y = .data[[variable]], color = "Outliers"),
      size = 1
    ) +
    ggplot2::geom_jitter(
      data = dplyr::filter(data, !idnummer %in% ids),
      alpha = 0.25,
      color = data_color,
      position = pos_jitter,
      show.legend = FALSE
    ) +
    ggplot2::geom_errorbar(
      data = mean_data,
      ggplot2::aes(x = x, ymin = ymin, ymax = ymax, color = "Mean (95% CI)"),
      width = 0.12, linewidth = 1.2,
      inherit.aes = FALSE
    ) +
    ggplot2::geom_point(
      data = mean_data,
      ggplot2::aes(x = x, y = y, color = "Mean (95% CI)"),
      size = 2, shape = 18,
      inherit.aes = FALSE
    ) +
    ggplot2::labs(title = NULL, x = NULL, y = beautify_names(variable), color = NULL) +
    ggplot2::scale_color_manual(
      values = c("Outliers" = outlier_color, "Mean (95% CI)" = ci_color)
    ) +
    garamond_theme
}

4 Detect Multivariate Outliers

Code 
# detect multivariate outliers using Local Outlier Factor (LOF)
# k is fixed at 20 to avoid the default combined-k behavior of bigutilsr::LOF,
# which aggregates across seq_k = c(4, 10, 30) and complicates interpretation
multivariate_outliers <- function(df, excl = NULL,
                                  id_var = "idnummer", k = 20L) {
  stopifnot(id_var %in% names(df))

  # add row numbers for tracking
  df_with_id <- df %>%
    dplyr::mutate(orig_row = dplyr::row_number())

  # prepare data: remove invariant columns, replace non-finite with NA
  df_clean <- df_with_id %>%
    remove_invariant() %>%
    dplyr::mutate(dplyr::across(
      dplyr::where(is.numeric),
      ~ ifelse(is.finite(.x), .x, NA)
    )) %>%
    dplyr::select(dplyr::where(~ is.numeric(.x) && sd(.x, na.rm = TRUE) != 0)) %>%
    dplyr::filter(complete.cases(.))

  df_clean <- numeric_df(df_clean)

  # keep id mapping separate from LOF feature set
  id_map <- df_clean %>%
    dplyr::select(dplyr::all_of(c(id_var, "orig_row")))

  # exclude identifiers and specified variables from LOF features
  excl_all <- unique(c(excl, id_var, "orig_row"))
  lof_df <- df_clean %>%
    dplyr::select(-dplyr::any_of(excl_all))

  if (ncol(lof_df) == 0) {
    stop("No numeric variables left for LOF after exclusions.")
  }

  # compute log-LOF with single, explicit k (Breunig et al., 2000).
  # with seq_k of length 1, `combine` is irrelevant because only one score
  # exists per observation; we keep the default `combine = max` for
  # compatibility with bigutilsr's API.
  llof <- bigutilsr::LOF(
    U = as.matrix(lof_df),
    seq_k = k,
    log = TRUE
  )

  # Tukey's adjusted boxplot rule (skew-corrected medcouple correction) is
  # the idiomatic bigutilsr cutoff for LOF output (see `?tukey_mc_up`).
  # Default alpha = 0.05 controls Type-I error after multiple-testing
  # correction; this is the canonical usage in the bigutilsr manual and is
  # retained here rather than tightening to the legacy thresh = 0.001,
  # which corresponds to a percentile rule, not a Type-I error.
  lof_cutoff <- bigutilsr::tukey_mc_up(llof)
  lof_outs_idxs <- which(llof > lof_cutoff)

  # map back to original ids
  out_rows <- id_map$orig_row[lof_outs_idxs]
  out_ids <- id_map[[id_var]][lof_outs_idxs]

  # filter out outliers by id
  df_without_outs <- df %>%
    dplyr::filter(!.data[[id_var]] %in% out_ids)

  list(
    df = df_without_outs,
    llof = llof,
    cutoff = lof_cutoff,
    k = k,
    id = out_ids,
    row = out_rows,
    n_features = ncol(lof_df),
    features_used = names(lof_df)
  )
}
