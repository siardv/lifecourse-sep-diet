2 Verify Parental Composite Integrity

# helper function to compute parental mean with consistent na.rm = FALSE logic
compute_parental_mean <- function(data, var1, var2) {
  # na.rm = FALSE is deliberate; see the rule note above verify-step5-data
  result <- rowMeans(data[, c(var1, var2)], na.rm = FALSE)
  both_missing <- is.na(data[[var1]]) & is.na(data[[var2]])
  result[both_missing] <- NA
  result
}

# verify education_parental computation
edu_components_present <- all(
  c("education_father", "education_mother") %in% names(imputed_data)
)

if ("education_parental" %in% names(imputed_data) && edu_components_present) {
  edu_expected <- compute_parental_mean(
    imputed_data, "education_father", "education_mother"
  )
  edu_match <- all.equal(
    imputed_data$education_parental,
    edu_expected,
    tolerance = 1e-10,
    check.attributes = FALSE
  )
  edu_status <- if (isTRUE(edu_match)) {
    if ("education_parental" %in% composites_created) {

      "<span style='color: blue;'>&#x2714; Recomputed from components</span>"
    } else {
      "<span style='color: green;'>&#x2714; Verified</span>"
    }
  } else {
    "<span style='color: red;'>&#x2718; Mismatch detected</span>"
  }
} else if (!"education_parental" %in% names(imputed_data)) {
  edu_status <- "<span style='color: red;'>&#x2718; Variable missing</span>"

6.2 Non-Partnered Respondents

Partner SES stays NA for non-partnered respondents in this dataset. Step 7 conditions these variables on partnership (zero for non-partnered) and enters them as interactions with the partner indicator, so every model retains the full sample while partner SES effects are identified by partnered respondents only.

# create comprehensive variable summary
var_summary <- data.frame(
  variable = names(imputed_data),
  stringsAsFactors = FALSE
) %>%
  dplyr::mutate(
    type = sapply(imputed_data, function(x) {
      if (is.numeric(x)) "Numeric" else if (is.factor(x)) "Factor" else class(x)[1]
    }),
    n_valid = sapply(imputed_data, function(x) sum(!is.na(x))),
    n_miss = sapply(imputed_data, function(x) sum(is.na(x))),
    pct_miss = round(100 * n_miss / n_obs, 1),
    n_unique = sapply(imputed_data, function(x) length(unique(x[!is.na(x)]))),
    category = dplyr::case_when(
      variable == "idnummer" ~ "Identifier",
      grepl("^education", variable) ~ "Education",
      grepl("^isei08", variable) ~ "Occupation (ISEI-08)",
      grepl("^isco08r", variable) ~ "Occupation (ISCO-08)",
      grepl("^brc14", variable) ~ "Occupation (BRC-14)",
      grepl("^dhd", variable) ~ "Dietary Outcome",
      grepl("income", variable) ~ "Income",
      grepl("partner", variable) ~ "Partner Status",
      TRUE ~ "Other"
    )
  ) %>%
  dplyr::arrange(category, variable) %>%
  dplyr::mutate(variable = beautify_names(variable))

var_summary %>%
  as_kable() %>%
